// define namespace 
resource "kubernetes_namespace" "my-web-app" {
  metadata {
    name = "my-web-app"
  }

  timeouts {
    delete = "20m"
  }
}

resource "kubernetes_deployment" "my_web_app__deployment" {

  metadata {
    name      = "deployment-my-web-app"
    namespace = "my-web-app"
  }

  spec {
    replicas = 2
    selector {
      match_labels = {
        "app.kubernetes.io/name" = "my-web-app"
      }
    }
    strategy {
      type = "RollingUpdate"

      rolling_update {
        max_surge       = "50%"
        max_unavailable = "50%"
      }
    }

    template {
      metadata {
        annotations = {}
        labels      = { "app.kubernetes.io/name" = "my-web-app" }
      }

      spec {

        node_selector                    = { "alpha.eksctl.io/nodegroup-name" = "ng1-mycluster1" }
        restart_policy                   = "Always"
        share_process_namespace          = false
        termination_grace_period_seconds = 30

        container {
          image = "nguyenquockhang1610/my-web-app:lastest"
          name  = "my-web-app"
          port {
            container_port = 80
            protocol       = "TCP"
          }

          resources {
          }
        }
      }
    }
  }
}

resource "kubernetes_service" "my-web-app__service" {

  metadata {
    name      = "service-my-web-app"
    namespace = "my-web-app"
  }

  spec {
    selector = {
      "app.kubernetes.io/name" = "my-web-app"
    }

    type = "NodePort"

    port {
      port        = 80
      protocol    = "TCP"
      target_port = "80"
    }
  }

}

resource "kubernetes_ingress" "my-web-app__ingress" {
  metadata {
    annotations = { "kubernetes.io/ingress.class" = "alb", "alb.ingress.kubernetes.io/scheme" = "internal", "alb.ingress.kubernetes.io/target-type" = "ip", "alb.ingress.kubernetes.io/listen-ports" = "[{\"HTTP\": 8080}]" }
    name        = "ingress-my-web-app"
    namespace   = "my-web-app"
  }

  spec {

    rule {
      http {
        path {
          path = "/*"
          backend {
            service_name = "service-my-web-app"
            service_port = "80"
          }
        }
      }
    }
  }
}
